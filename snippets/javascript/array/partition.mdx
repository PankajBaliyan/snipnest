```js
function partition(arr, callback) {
  return arr.reduce(
    ([pass, fail], elem) => {
      if (callback(elem)) {
        pass.push(elem);
      } else {
        fail.push(elem);
      }
      return [pass, fail];
    },
    [[], []]
  );
}
```

```js
const arr = [1, 2, 3, 4, 5];
const isEven = (n) => n % 2 === 0;
partition(arr, isEven); // [[2, 4], [1, 3, 5]]
```
