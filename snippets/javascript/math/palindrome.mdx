```javascript
function isNumericPalindrome(num) {
  // Convert the number to a string
  const str = num.toString();

  // Check if the string is equal to its reverse
  return str === str.split("").reverse().join("");
}
```

```javascript
isNumericPalindrome(12345); // false
```
