```javascript
function rgbToHex(r, g, b) {
  return `#${((1 << 24) | (r << 16) | (g << 8) | b)
    .toString(16)
    .slice(1)
    .toUpperCase()}`;
}
```

```javascript
rgbToHex(255, 99, 71); // #ff6347
```
